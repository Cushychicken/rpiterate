kernel=zImage

# To use an external initramfs file
#initramfs rootfs.cpio.gz

# Disable overscan assuming the display supports displaying the full resolution
# If the text shown on the screen disappears off the edge, comment this out
disable_overscan=1

# How much memory in MB to assign to the GPU on Pi models having
# 256, 512 or 1024 MB total memory
gpu_mem_256=100
gpu_mem_512=100
gpu_mem_1024=100

# Enable UART0 for serial console on ttyAMA0
dtoverlay=miniuart-bt

# Enable I2C for touchscreen (most touchscreens use I2C)
dtparam=i2c_arm=on

# Enable KMS (Kernel Mode Setting) - required for modern display support
# Use vc4-fkms-v3d for better compatibility with official touchscreen
dtoverlay=vc4-fkms-v3d

# Official Raspberry Pi 7" touchscreen
# The touchscreen is connected via DSI and uses I2C for touch input
# No additional overlay needed - the touchscreen is detected automatically when vc4-fkms-v3d is loaded
